table checksum

uint32 CalcTableChecksum(uint32 *table, uint32 numberOfBytesInTable){
    uint32 sum = 0;
    uint32 nLongs = (numberOfBytesInTable + 3) / 4;
    while (nLongs-- > 0){
        sum += *table++;
    }
    return sum;
}
